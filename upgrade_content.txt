#### 8. Build the PainPointWidget Component
Create `app/components/PainPointWidget.tsx`:
```tsx
'use client';
import { useState } from 'react';
interface PainPoint {
  pain: string;
  severity?: string;
  source?: string;
}
export default function PainPointWidget({ webhookUrl }: { webhookUrl: string }) {
  const [loading, setLoading] = useState(false);
  const [results, setResults] = useState<PainPoint[]>([]);
  const [error, setError] = useState('');
  const scan = async () => {
    setLoading(true);
    setError('');
    setResults([]);
    try {
      const res = await fetch(webhookUrl, { method: 'GET' });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const data = await res.json();
      // Assume data is an array of pain points
      setResults(Array.isArray(data) ? data : []);
    } catch (err: any) {
      setError(err.message || 'Failed to scan');
    } finally {
      setLoading(false);
    }
  };
  return (
    <div className="my-8 p-6 border rounded-xl bg-gray-50 dark:bg-gray-800">
      <button
        onClick={scan}
        disabled={loading}
        className="px-6 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 disabled:opacity-50"
      >
        {loading ? 'Scanning...' : '≡ƒöì Scan Latest Pain Points'}
      </button>
      {loading && <p className="mt-4 text-gray-500">Analyzing Reddit posts with DeepSeek (15-20 sec)...</p>}
      {error && <p className="mt-4 text-red-600">Error: {error}</p>}
      {results.length > 0 && (
        <div className="mt-6">
          <h3 className="text-xl font-semibold mb-3">Identified Pain Points</h3>
          <ul className="space-y-3">
            {results.map((p, idx) => (
              <li key={idx} className="border-l-4 border-blue-500 pl-4">
                <p>{p.pain}</p>
                {p.severity && <span className="text-sm text-gray-500">Severity: {p.severity}</span>}
              </li>
            ))}
          </ul>
        </div>
      )}
    </div>
  );
}
```

But since `require` might not work with static export, an alternative: Use a bundled approach with `next-mdx-remote`. However, to keep it simple for initial deployment, we can just use `.mdx` files imported statically. For Cloudflare static export, a better method is to list all posts in `generateStaticParams` and read the file content at build time using `fs` (only during build). I will provide a build-safe implementation:
**Alternative robust MDX handling (recommended):**  
Create `app/blog/[slug]/page.tsx` that uses `next-mdx-remote`. Install:
```bash
npm install next-mdx-remote
```
Then:
```tsx
import { notFound } from 'next/navigation';
import { readFileSync } from 'fs';
import path from 'path';
import { MDXRemote } from 'next-mdx-remote/rsc';
import { blogPosts } from '../posts';
import PainPointWidget from '@/app/components/PainPointWidget';
const components = { PainPointWidget }; // pass the widget as a component to MDX
export async function generateStaticParams() {
  return blogPosts.map((post) => ({ slug: post.slug }));
}
export default async function BlogPost({ params }: { params: { slug: string } }) {
  const { slug } = params;
  const filePath = path.join(process.cwd(), 'contents', 'blog', `${slug}.mdx`);
  let source: string;
  try {
    source = readFileSync(filePath, 'utf-8');
  } catch {
    notFound();
  }
  return (
    <div className="container mx-auto px-4 py-12 max-w-3xl">
      <article className="prose prose-lg dark:prose-invert">
        <MDXRemote source={source} components={components} />
      </article>
    </div>
  );
}
```
This works because `readFileSync` runs at build time. It's safe for static export.
